64 std::vector<ComPWA::Tools::IntensityComponent>
65 IntensityBuilderXML::createIntensityComponents(
66  std::vector<std::vector<std::string>> ComponentList) {
67  LOG(TRACE) << "Creating intensity components...";
68 
69  if (UniqueComponentFTMapping.size() == 0) {
70  LOG(INFO) << "Components map is empty. Creating full Intensity first.";
71  createIntensity();
72  }
73 
74  // then put these FT together according to the string vectors
75  std::vector<ComPWA::Tools::IntensityComponent> IntensityComponents;
76 
77  for (auto const &Component : ComponentList) {
78  std::string ComponentName;
79  std::map<
80  std::string,
81  std::pair<std::string, std::shared_ptr<ComPWA::FunctionTree::TreeNode>>>
82  NewUniqueComponentFTMapping;
83  std::string Type("");
84  std::vector<std::shared_ptr<ComPWA::FunctionTree::TreeNode>> FTList;
85  for (auto const &x : Component) {
86  auto FindResult = UniqueComponentFTMapping.find(x);
87  if (UniqueComponentFTMapping.end() != FindResult) {
88  ComponentName += "_" + x;
89  if (Type == "") {
90  Type = FindResult->second.first;
91  } else {
92  if (Type != FindResult->second.first) {
93  LOG(ERROR) << "Component " << x
94  << " incompatible with previous type " << Type
95  << " Skipping...";
96  continue;
97  }
98  }
99  NewUniqueComponentFTMapping.insert(*FindResult);
100  FTList.push_back(FindResult->second.second);
101  } else {
102  LOG(ERROR) << "Component " << x << " not found! Skipping...";
103  }
104  }
105  if (FTList.size() == 0)
106  continue;
107 
108  ComponentName.erase(0, 1);
109 
110  LOG(INFO) << "Building component " << ComponentName;
111 
112  // empty component were already removed so [0] is safe
113  if (Type == "Amplitude") {
114  IntensityComponents.push_back(std::make_pair(
115  ComponentName,
116  std::make_shared<ComPWA::FunctionTree::FunctionTreeIntensity>(
117  createCoherentIntensityFT(FTList), Parameters, Data.Data)));
118  LOG(INFO) << "as a CoherentIntensity";
119  } else {
120  IntensityComponents.push_back(std::make_pair(
121  ComponentName,
122  std::make_shared<ComPWA::FunctionTree::FunctionTreeIntensity>(
123  createIncoherentIntensityFT(FTList), Parameters, Data.Data)));
124  LOG(INFO) << "as a IncoherentIntensity";
125  }
126  }
127  return IntensityComponents;
128 }

342 std::shared_ptr<ComPWA::FunctionTree::TreeNode>
343 IntensityBuilderXML::createIntegrationStrategyFT(
344  std::shared_ptr<ComPWA::FunctionTree::TreeNode> UnnormalizedIntensity,
345  std::shared_ptr<ComPWA::FunctionTree::Value<std::vector<double>>>
346  PhspWeights,
347  double PhspWeightSum, std::string IntegratorClassName) {
348  LOG(TRACE) << "creating IntegrationStrategy ...";
349 
350  using namespace ComPWA::FunctionTree;
351  std::shared_ptr<TreeNode> tr;
352 
353  if (IntegratorClassName == "MCIntegrationStrategy") {
354  // update the PhspData container
355  updateDataContainerState();
356 
357  tr = std::make_shared<TreeNode>(
358  ValueFactory(ParType::DOUBLE),
359  std::shared_ptr<Strategy>(new Inverse(ParType::DOUBLE)));
360  auto Integral = std::make_shared<TreeNode>(
361  std::shared_ptr<Strategy>(new MultAll(ParType::DOUBLE)));
362  tr->addNode(Integral);
363  Integral->addNodes(
364  {createLeaf(Kinematic.phspVolume(), "PhspVolume"),
365  createLeaf(1.0 / PhspWeightSum, "Inverse_PhspWeightSum")});
366  auto Sum = std::make_shared<TreeNode>(
367  std::shared_ptr<Strategy>(new AddAll(ParType::DOUBLE)));
368  Integral->addNode(Sum);
369  auto WeightedIntensities = std::make_shared<TreeNode>(
370  std::shared_ptr<Strategy>(new MultAll(ParType::MDOUBLE)));
371  Sum->addNode(WeightedIntensities);
372 
373  if (PhspWeights)
374  WeightedIntensities->addNode(createLeaf(PhspWeights));
375  WeightedIntensities->addNode(UnnormalizedIntensity);
376  } else {
377  LOG(WARNING) << "IntensityBuilderXML::createIntegrationStrategyFT(): "
378  "IntegrationStrategy type "
379  << IntegratorClassName << " unknown!";
380  }
381 
382  return tr;
383 }

485 std::shared_ptr<ComPWA::FunctionTree::TreeNode>
486 IntensityBuilderXML::createCoefficientAmplitudeFT(
487  const boost::property_tree::ptree &pt,
488  const ComPWA::FunctionTree::ParameterList &DataSample) {
489  LOG(TRACE) << "constructing CoefficientAmplitude ...";
490 
491  std::shared_ptr<FitParameter> Magnitude(nullptr);
492  std::shared_ptr<FitParameter> Phase(nullptr);
493  auto PreFactor = std::complex<double>(1, 0);
494  bool IsPreFactorSet(false);
495  boost::property_tree::ptree AmpPT;
496  for (const auto &v : pt) {
497  if (v.first == "Parameter") {
498  if (v.second.get<std::string>("<xmlattr>.Type") == "Magnitude")
499  Magnitude = std::make_shared<FitParameter>(v.second);
500  if (v.second.get<std::string>("<xmlattr>.Type") == "Phase")
501  Phase = std::make_shared<FitParameter>(v.second);
502  } else if (v.first == "PreFactor") {
503  IsPreFactorSet = true;
504  boost::optional<double> optr =
505  v.second.get_optional<double>("<xmlattr>.Magnitude");
506  if (optr.is_initialized()) {
507  double r(optr.value());
508  if (r < 0.0)
509  throw BadConfig(
510  "IntensityBuilderXML::createCoefficientAmplitudeFT() | "
511  "PreFactor Magnitude below zero!");
512  double p(0.0);
513  boost::optional<double> optp =
514  v.second.get_optional<double>("<xmlattr>.Phase");
515  if (optp.is_initialized())
516  p = optp.value();
517  PreFactor = std::polar(r, p);
518  } else {
519  double real = v.second.get<double>("<xmlattr>.Real");
520  double im(0.0);
521  boost::optional<double> optim =
522  v.second.get_optional<double>("<xmlattr>.Imaginary");
523  if (optim.is_initialized())
524  im = optim.value();
525  PreFactor = std::complex<double>(real, im);
526  }
527  } else if (v.first == "Amplitude") {
528  AmpPT = v.second;
529  } else if (v.first != "<xmlattr>") {
530  throw BadConfig("IntensityBuilderXML::createCoefficientAmplitudeFT() | "
531  "Unknown tag " +
532  v.first + "!");
533  }
534  }
535 
536  auto amp_ft = createAmplitudeFT(AmpPT, DataSample);
537 
538  auto tr = std::make_shared<ComPWA::FunctionTree::TreeNode>(
539  ComPWA::FunctionTree::MComplex("", 0),
540  std::make_shared<ComPWA::FunctionTree::MultAll>(
541  ComPWA::FunctionTree::ParType::MCOMPLEX));
542 
543  bool AddStrength(true);
544  if (IsPreFactorSet) {
545  tr->addNode(FunctionTree::createLeaf(PreFactor, "PreFactor"));
546  if (!Magnitude && !Phase) {
547  AddStrength = false;
548  } else if (!Magnitude && Phase) {
549  throw BadParameter("IntensityBuilderXML::createCoefficientAmplitude() | "
550  "No magnitude parameter found while phase is set.");
551  } else if (!Phase && Magnitude)
552  throw BadParameter("IntensityBuilderXML::createCoefficientAmplitude() | "
553  "No phase parameter found, while magnitude is set.");
554  } else {
555  if (!Magnitude)
556  throw BadParameter("IntensityBuilderXML::createCoefficientAmplitude() | "
557  "No magnitude parameter found and no prefactor set.");
558  if (!Phase)
559  throw BadParameter("IntensityBuilderXML::createCoefficientAmplitude() | "
560  "No phase parameter found, while magnitude is set.");
561  }
562 
563  if (AddStrength) {
564  Magnitude = Parameters.addUniqueParameter(Magnitude);
565  Phase = Parameters.addUniqueParameter(Phase);
566 
567  auto Strength = std::make_shared<ComPWA::FunctionTree::TreeNode>(
568  std::make_shared<ComPWA::FunctionTree::Value<std::complex<double>>>(),
569  std::make_shared<ComPWA::FunctionTree::Complexify>(
570  ComPWA::FunctionTree::ParType::COMPLEX));
571  Strength->addNodes({createLeaf(Magnitude), createLeaf(Phase)});
572 
573  tr->addNode(Strength);
574  }
575  tr->addNode(amp_ft);
576 
577  return tr;
578 }

622  auto InvMassSq =
623  FunctionTree::findMDoubleValue(std::get<0>(KinVarNames), DataSample);
624  auto Theta =
625  FunctionTree::findMDoubleValue(std::get<1>(KinVarNames), DataSample);
626  auto Phi =
627  FunctionTree::findMDoubleValue(std::get<2>(KinVarNames), DataSample);
628 
629  std::string name = pt.get<std::string>("DecayParticle.<xmlattr>.Name");
630  auto partProp = ComPWA::findParticle(PartList, name);
631 
632  double J = partProp.getQuantumNumber<double>("Spin");
633  double mu(pt.get<double>("DecayParticle.<xmlattr>.Helicity"));
634  // if the node OrbitalAngularMomentum does not exist, set it to spin J as
635  // default value
636  unsigned int L(J);
637 
638  auto Mass = std::make_shared<FunctionTree::FitParameter>(
639  partProp.getMass().Name, partProp.getMass().Value,
640  partProp.getMass().Error.first);
641  Mass->fixParameter(partProp.getMass().IsFixed);
642  Mass = Parameters.addUniqueParameter(Mass);
643 
644  double PreFactor(1.0);
645  const auto &canoSum = pt.get_child_optional("CanonicalSum");
646  if (canoSum) {
647  const auto &sumTree = canoSum.get();
648  L = sumTree.get<unsigned int>("<xmlattr>.L");
649  double coef = std::sqrt((2.0 * L + 1) / (2 * J + 1));
650  for (const auto &cg : sumTree.get_child("")) {
651  if (cg.first != "ClebschGordan")
652  continue;
653  double j1 = cg.second.get<double>("<xmlattr>.j1");
654  double m1 = cg.second.get<double>("<xmlattr>.m1");
655  double j2 = cg.second.get<double>("<xmlattr>.j2");
656  double m2 = cg.second.get<double>("<xmlattr>.m2");
657  double J = cg.second.get<double>("<xmlattr>.J");
658  double M = cg.second.get<double>("<xmlattr>.M");
659  coef *= ComPWA::QFT::Clebsch(j1, m1, j2, m2, J, M);
660  }
661  PreFactor *= coef;
662  }
663 
664  // Read name and helicities from decay products
665  auto decayProducts = pt.get_child("DecayProducts");
666  if (decayProducts.size() != 2)
667  throw boost::property_tree::ptree_error(
668  "IntensityBuilderXML::createHelicityDecayFT(): Expect exactly two "
669  "decay products (" +
670  std::to_string(decayProducts.size()) + " given)!");
671 
672  std::pair<std::string, std::string> DecayProducts;
673  std::pair<double, double> DecayHelicities;
674 
675  auto p = decayProducts.begin();
676  DecayProducts.first = p->second.get<std::string>("<xmlattr>.Name");
677  DecayHelicities.first = p->second.get<double>("<xmlattr>.Helicity");
678  ++p;
679  DecayProducts.second = p->second.get<std::string>("<xmlattr>.Name");
680  DecayHelicities.second = p->second.get<double>("<xmlattr>.Helicity");
681 
682  auto parMass1 = std::make_shared<FitParameter>(
683  ComPWA::findParticle(PartList, DecayProducts.first).getMass().Name,
684  ComPWA::findParticle(PartList, DecayProducts.first).getMass().Value);
685  auto parMass2 = std::make_shared<FitParameter>(
686  ComPWA::findParticle(PartList, DecayProducts.second).getMass().Name,
687  ComPWA::findParticle(PartList, DecayProducts.second).getMass().Value);
688  parMass1->fixParameter(
689  ComPWA::findParticle(PartList, DecayProducts.first).getMass().IsFixed);
690  parMass2->fixParameter(
691  ComPWA::findParticle(PartList, DecayProducts.second).getMass().IsFixed);
692  parMass1 = Parameters.addUniqueParameter(parMass1);
693  parMass2 = Parameters.addUniqueParameter(parMass2);
694  auto decayInfo = partProp.getDecayInfo();
695  std::string FFType("");
696  std::shared_ptr<Dynamics::FormFactor> FormFactor =
697  std::make_shared<Dynamics::NoFormFactor>();
698  std::shared_ptr<FitParameter> parRadius;
699  std::shared_ptr<FitParameter> Width;
700  for (const auto &node : decayInfo.get_child("")) {
701  if (node.first == "FormFactor") {
702  FFType = node.second.get<std::string>("<xmlattr>.Type");
703  if ("BlattWeisskopf" == FFType) {
704  if (L > 4)
705  throw std::runtime_error(
706  "createHelicityDecayFT() | Blatt-Weisskopf form factors are "
707  "implemented for L up to 4!");
708  FormFactor = std::make_shared<Dynamics::BlattWeisskopfFormFactor>();
709  } else if ("CrystalBarrel" == FFType) {
710  if (L != 0)
711  throw std::runtime_error(
712  "createHelicityDecayFT() | Crystal Barrel form factors are "
713  "implemented for L=0 only!");
714  FormFactor = std::make_shared<Dynamics::CrystalBarrelFormFactor>();
715  } else {
716  // if user specifies unknown form factor
717  throw std::runtime_error("createHelicityDecayFT() | Form factor type " +
718  FFType + " not specified!");
719  }
720  } else if (node.first == "Parameter") {
721  std::string parType = node.second.get<std::string>("<xmlattr>.Type");
722  if (parType == "Width") {
723  Width = std::make_shared<FitParameter>(node.second);
724  Width = Parameters.addUniqueParameter(Width);
725  } else if (parType == "MesonRadius") {
726  parRadius = std::make_shared<FitParameter>(node.second);
727  parRadius = Parameters.addUniqueParameter(parRadius);
728  }
729  }
730  }
731 
732  std::string decayType = partProp.getDecayType();
733 
734  std::shared_ptr<ComPWA::FunctionTree::TreeNode> DynamicFunctionFT(nullptr);
735 
736  if (decayType == "stable") {
737  throw std::runtime_error(
738  "IntensityBuilderXML::createHelicityDecayFT(): Stable particle is "
739  "given as mother particle of a decay. Makes no sense!");
740  } else if (decayType.find("BreitWigner") != std::string::npos) {
741  using namespace ComPWA::Physics::Dynamics::RelativisticBreitWigner;
742  InputInfo RBW;
743  RBW.Type = decayType;
744  RBW.Mass = Mass;
745  RBW.Width = Width;
746  RBW.MesonRadius = parRadius;
747  RBW.DaughterMasses = std::make_pair(parMass1, parMass2);
748  RBW.FormFactorFunctor = FormFactor;
749  RBW.L = L;
750  DynamicFunctionFT = createFunctionTree(RBW, InvMassSq);
751  } else if (decayType == "flatte") {
752  ComPWA::Physics::Dynamics::Flatte::InputInfo FlatteInfo;
753  FlatteInfo.Mass = Mass;
754  FlatteInfo.MesonRadius = parRadius;
755  FlatteInfo.DaughterMasses = std::make_pair(parMass1, parMass2);
756  FlatteInfo.FormFactorFunctor = FormFactor;
757  FlatteInfo.L = L;
758  std::vector<Dynamics::Coupling> couplings;
759  // Read parameters
760  for (const auto &v : decayInfo.get_child("")) {
761  if (v.first != "Parameter")
762  continue;
763  std::string type = v.second.get<std::string>("<xmlattr>.Type");
764  if (type == "Coupling") {
765  auto c = Dynamics::Coupling(PartList, v.second);
766  c.G = Parameters.addUniqueParameter(c.G);
767 
768  if ((c.MassA->value() == parMass1->value() &&
769  c.MassB->value() == parMass2->value()) ||
770  (c.MassB->value() == parMass1->value() &&
771  c.MassA->value() == parMass2->value())) {
772  FlatteInfo.G = c.G;
773  } else {
774  c.MassA = Parameters.addUniqueParameter(c.MassA);
775  c.MassB = Parameters.addUniqueParameter(c.MassB);
776  couplings.push_back(c);
777  }
778  }
779  }
780  FlatteInfo.HiddenCouplings = couplings;
781  DynamicFunctionFT = createFunctionTree(FlatteInfo, InvMassSq);
782  } else if (decayType == "voigt") {
783  using namespace ComPWA::Physics::Dynamics::Voigtian;
784  InputInfo VoigtInfo;
785  VoigtInfo.Mass = Mass;
786  VoigtInfo.Width = Width;
787  VoigtInfo.MesonRadius = parRadius;
788  VoigtInfo.DaughterMasses = std::make_pair(parMass1, parMass2);
789  VoigtInfo.FormFactorFunctor = FormFactor;
790  VoigtInfo.L = L;
791  VoigtInfo.Sigma = decayInfo.get<double>("Resolution.<xmlattr>.Sigma");
792  DynamicFunctionFT = createFunctionTree(VoigtInfo, InvMassSq);
793  } else if (decayType == "virtual" || decayType == "nonResonant") {
794  DynamicFunctionFT = FunctionTree::createLeaf(1, "Dynamics");
795  } else {
796  throw std::runtime_error("HelicityDecay::Factory() | Unknown decay type " +
797  decayType + "!");
798  }
799 
800  auto AngularFunction =
801  ComPWA::Physics::HelicityFormalism::WignerD::createFunctionTree(
802  J, mu, DecayHelicities.first - DecayHelicities.second, Theta, Phi);
803 
804  using namespace ComPWA::FunctionTree;
805  auto tr = std::make_shared<ComPWA::FunctionTree::TreeNode>(
806  MComplex("", 0), std::make_shared<MultAll>(ParType::MCOMPLEX));
807  tr->addNodes(
808  {createLeaf(PreFactor, "PreFactor"), AngularFunction, DynamicFunctionFT});
809 
810  // set production formfactor
811  if (FFType != "" && L > 0) {
812  if (parRadius == nullptr) {
813  throw std::runtime_error("IntensityBuilderXML::createHelicityDecayFT(): "
814  "No MesonRadius given for amplitude " +
815  name +
816  "! It is needed to calculate the form factor!");
817  }
818 
819  std::shared_ptr<ComPWA::FunctionTree::TreeNode> ProductionFormFactorFT =
820  Dynamics::createFunctionTree(parMass1, parMass2, parRadius, L,
821  FormFactor, InvMassSq);
822 
823  tr->addNode(ProductionFormFactorFT);
824  }
825 
826  return tr;
827 }
