76 inline std::complex<double>
77 relativisticBreitWigner(double mSq, double mR, double ma, double mb,
78  double width, unsigned int L, double mesonRadius,
79  std::shared_ptr<FormFactor> FormFactorFunctor) {
80 
81  std::complex<double> i(0, 1);
82  double sqrtS = std::sqrt(mSq);
83 
84  // Phase space factors at sqrt(s) and at the resonance position
85  auto phspFactorSqrtS = phspFactor(sqrtS, ma, mb);
86  auto phspFactormR = phspFactor(mR, ma, mb);
87 
88  // Check if we have an event which is exactly at the phase space boundary
89  if (phspFactorSqrtS == std::complex<double>(0, 0))
90  return std::complex<double>(0, 0);
91 
92  // The each FormFactor includes a term q^L. Therefore only q instead of
93  // q^(2L+1) has to be calculated. Also because phspFactor ~ q/sqrt(s) (see
94  // PDG2018, equation 48.2) the factor \frac{M_R}{\sqrt{s}} can also be omitted
95  std::complex<double> qRatio = (phspFactorSqrtS / phspFactormR);
96 
97  double ffR =
98  FormFactorFunctor->operator()(qSquared(mR * mR, ma, mb), L, mesonRadius);
99  double ff =
100  FormFactorFunctor->operator()(qSquared(mSq, ma, mb), L, mesonRadius);
101  std::complex<double> barrierTermSq = qRatio * (ff * ff) / (ffR * ffR);
102 
103  std::complex<double> denom(mR * mR - mSq, 0);
104  denom += (-1.0) * i * sqrtS * (width * barrierTermSq);
105 
106  std::complex<double> result = ff / denom;
107 
108  assert((!std::isnan(result.real()) || !std::isinf(result.real())) &&
109  "RelativisticBreitWigner::standardRelBW() | Result is NaN or Inf!");
110  assert((!std::isnan(result.imag()) || !std::isinf(result.imag())) &&
111  "RelativisticBreitWigner::standardRelBW() | Result is NaN or Inf!");
112 
113  return result;
114 }

146 inline std::complex<double> relativisticBreitWignerAnalyticCont(
147  double mSq, double mR, double ma, double mb, double width, unsigned int L,
148  double mesonRadius, std::shared_ptr<FormFactor> FormFactorFunctor) {
149 
150  std::complex<double> i(0, 1);
151  double sqrtS = std::sqrt(mSq);
152 
153  // Phase space factors at sqrt(s) and at the resonance position
154  auto phspFactorSqrtS = phspFactorAC(sqrtS, ma, mb);
155  auto phspFactormR = phspFactorAC(mR, ma, mb);
156 
157  // Check if we have an event which is exactly at the phase space boundary
158  if (phspFactorSqrtS == std::complex<double>(0, 0))
159  return std::complex<double>(0, 0);
160 
161  // The each FormFactor includes a term q^L. Therefore only q instead of
162  // q^(2L+1) has to be calculated. Also because phspFactor ~ q/sqrt(s) (see
163  // PDG2018, equation 48.2) the factor \frac{M_R}{\sqrt{s}} can also be omitted
164  std::complex<double> qRatio = (phspFactorSqrtS / phspFactormR);
165 
166  double ffR = FormFactorFunctor->operator()(
167  std::pow(std::abs(qValueAC(mR, ma, mb)), 2), L, mesonRadius);
168  double ff = FormFactorFunctor->operator()(
169  std::pow(std::abs(qValueAC(sqrtS, ma, mb)), 2), L, mesonRadius);
170  std::complex<double> barrierTermSq = qRatio * (ff * ff) / (ffR * ffR);
171 
172  // Calculate normalized vertex function gammaA(s_R) at the resonance position
173  // (see PDG2018, Chapter 48.2)
174  std::complex<double> gammaA(1, 0); // spin==0
175  if (L > 0) {
176  std::complex<double> qR = std::pow(qValueAC(mR, ma, mb), L);
177  gammaA = ffR * qR;
178  }
179 
180  // Coupling to the final state (ma, mb)
181  std::complex<double> g_final =
182  widthToCoupling(mR, width, gammaA, phspFactorSqrtS);
183 
184  std::complex<double> denom(mR * mR - mSq, 0);
185  denom += (-1.0) * i * sqrtS * (width * barrierTermSq);
186 
187  std::complex<double> result = g_final / denom;
188 
189  assert((!std::isnan(result.real()) || !std::isinf(result.real())) &&
190  "RelativisticBreitWigner::relativisticBreitWignerAnalyticCont() | "
191  "Result is NaN or Inf!");
192  assert((!std::isnan(result.imag()) || !std::isinf(result.imag())) &&
193  "RelativisticBreitWigner::relativisticBreitWignerAnalyticCont() | "
194  "Result is NaN or Inf!");
195 
196  return result;
197 }
