55 HelicityKinematics::reduceToPhaseSpace(const std::vector<Event> &Events) const {
56  std::vector<Event> tmp;
57  LOG(INFO) << "HelicityKinematics::reduceToPhaseSpace(): "
58  "Remove all events outside PHSP boundary from data sample.";
59 
60  std::copy_if(
61  Events.begin(), Events.end(), std::back_inserter(tmp), [this](auto evt) {
62  for (auto const &x : InvariantMassesSquared) {
63  auto bounds = InvMassBounds.at(x.second);
64  double mass = this->calculateInvariantMassSquared(evt, x.first);
65  if (mass < bounds.first || mass > bounds.second)
66  return false;
67  }
68  for (auto const &x : Subsystems) {
69  auto angles = this->calculateHelicityAngles(evt, x.first);
70  if (angles.first < 0 || angles.first > M_PI)
71  return false;
72  if (angles.second < -M_PI || angles.second > M_PI)
73  return false;
74  }
75  return true;
76  });
77 
78  LOG(INFO) << "reduceToPhaseSpace(): Removed " << Events.size() - tmp.size()
79  << " from " << Events.size() << " Events ("
80  << (1.0 - tmp.size() / Events.size()) * 100 << "%).";
81 
82  return tmp;
83 }

85 std::pair<double, double>
86 HelicityKinematics::calculateHelicityAngles(const Event &Event,
87  const SubSystem &SubSys) const {
88  FourMomentum FinalA, FinalB;
89  for (auto s : SubSys.getFinalStates().at(0)) {
90  unsigned int index = KinematicsInfo.convertFinalStateIDToPositionIndex(s);
91  FinalA += Event.ParticleList[index].fourMomentum();
92  }
93 
94  for (auto s : SubSys.getFinalStates().at(1)) {
95  unsigned int index = KinematicsInfo.convertFinalStateIDToPositionIndex(s);
96  FinalB += Event.ParticleList[index].fourMomentum();
97  }
98 
99  // Four momentum of the decaying resonance
100  FourMomentum State = FinalA + FinalB;
101 
102  QFT::Vector4<double> DecayingState(State);
103  QFT::Vector4<double> Daughter(FinalA);
104 
105  // calculate the recoil and parent recoil
106  auto const &RecoilState = SubSys.getRecoilState();
107 
108  // the first step is boosting everything into the rest system of the
109  // decaying state
110  Daughter.Boost(DecayingState);
111 
112  if (RecoilState.size() > 0) {
113  FourMomentum TempRecoil;
114  for (auto s : RecoilState) {
115  unsigned int index = KinematicsInfo.convertFinalStateIDToPositionIndex(s);
116  TempRecoil += Event.ParticleList[index].fourMomentum();
117  }
118  QFT::Vector4<double> Recoil(TempRecoil);
119 
120  Recoil.Boost(DecayingState);
121 
122  // rotate recoil so that recoil points in the -z direction
123  Daughter.RotateZ(-Recoil.Phi());
124  Daughter.RotateY(M_PI - Recoil.Theta());
125 
126  auto const &ParentRecoilState = SubSys.getParentRecoilState();
127  // in case there is no parent recoil, it is artificially along z
128  QFT::Vector4<double> ParentRecoil(0.0, 0.0, 0.0, 1.0);
129  if (ParentRecoilState.size() > 0) {
130  FourMomentum TempParentRecoil;
131  for (auto s : ParentRecoilState) {
132  unsigned int index =
133  KinematicsInfo.convertFinalStateIDToPositionIndex(s);
134  TempParentRecoil += Event.ParticleList[index].fourMomentum();
135  }
136  ParentRecoil = TempParentRecoil;
137  }
138 
139  ParentRecoil.Boost(DecayingState);
140  ParentRecoil.RotateZ(-Recoil.Phi());
141  ParentRecoil.RotateY(M_PI - Recoil.Theta());
142 
143  // rotate around the z-axis so that the parent recoil lies in the x-z
144  // plane
145  Daughter.RotateZ(M_PI - ParentRecoil.Phi());
146  }
147 
148  double cosTheta = Daughter.CosTheta();
149  double phi = Daughter.Phi();
150 
151  return std::make_pair(std::acos(cosTheta), phi);
152 }

187  Dataset.Data.at(x.second.second).push_back(Angles.second);
188  }
189  }
190  for (auto const &event : Events) {
191  Dataset.Weights.push_back(event.Weight);
192  }
193  return Dataset;

226 std::vector<std::pair<ComPWA::IndexList, ComPWA::IndexList>>
227 redistributeIndexLists(const ComPWA::IndexList &A, const ComPWA::IndexList &B) {
228  using ComPWA::IndexList;
229  std::vector<std::pair<IndexList, IndexList>> NewIndexLists;
230  if (A.size() < B.size())
231  throw std::runtime_error("HelicityKinematics::redistributeIndexLists(): A "
232  "cannot have less content than B!");
233  if (A.size() == 2 && B.size() == 0) {
234  NewIndexLists.push_back(std::make_pair(IndexList{A[0]}, IndexList{A[1]}));
235  } else if (A.size() - B.size() > 1) {
236  for (unsigned int i = 0; i < A.size(); ++i) {
237  IndexList TempB(B);
238  TempB.push_back(A[i]);
239  IndexList TempA(A);
240  TempA.erase(TempA.begin() + i);
241  NewIndexLists.push_back(std::make_pair(TempA, TempB));
242  }
243  }
244  return NewIndexLists;
245 }

249 IndexListTuple sortSubsystem(const IndexListTuple &SubSys) {
250  IndexList FinalStateA(std::get<0>(SubSys));
251  IndexList FinalStateB(std::get<1>(SubSys));
252  IndexList RecoilState(std::get<2>(SubSys));
253  IndexList ParentRecoilState(std::get<3>(SubSys));
254 
255  // create sorted entry
256  std::sort(FinalStateA.begin(), FinalStateA.end());
257  std::sort(FinalStateB.begin(), FinalStateB.end());
258  std::sort(RecoilState.begin(), RecoilState.end());
259  std::sort(ParentRecoilState.begin(), ParentRecoilState.end());
260 
261  IndexListTuple SortedTuple;
262  if (FinalStateA > FinalStateB)
263  SortedTuple = std::make_tuple(FinalStateB, FinalStateA, RecoilState,
264  ParentRecoilState);
265  else
266  SortedTuple = std::make_tuple(FinalStateA, FinalStateB, RecoilState,
267  ParentRecoilState);
268  return SortedTuple;
269 }

271 void HelicityKinematics::createAllSubsystems() {
272  std::vector<IndexListTuple> CurrentSubsystems;
273  LOG(INFO) << "creating all Subsystems!";
274 
275  std::vector<IndexListTuple> AllSubsystems;
276  // add current subsystems
277  for (auto const &SubSys : Subsystems) {
278  AllSubsystems.push_back(
279  std::make_tuple(KinematicsInfo.convertFinalStateIDToPositionIndex(
280  SubSys.first.getFinalStates()[0]),
281  KinematicsInfo.convertFinalStateIDToPositionIndex(
282  SubSys.first.getFinalStates()[1]),
283  KinematicsInfo.convertFinalStateIDToPositionIndex(
284  SubSys.first.getRecoilState()),
285  KinematicsInfo.convertFinalStateIDToPositionIndex(
286  SubSys.first.getParentRecoilState())));
287  }
288 
289  IndexList FinalStateIDs(KinematicsInfo.getFinalStateParticleCount());
290  unsigned int i = 0;
291  std::generate(FinalStateIDs.begin(), FinalStateIDs.end(),
292  [&i]() -> unsigned int { return i++; });
293  for (auto const &x : redistributeIndexLists(FinalStateIDs, IndexList())) {
294  CurrentSubsystems.push_back(
295  std::make_tuple(x.first, x.second, IndexList(), IndexList()));
296  }
297 
298  while (CurrentSubsystems.size() > 0) {
299  auto TempSubsystems(CurrentSubsystems);
300  CurrentSubsystems.clear();
301  for (auto const &x : TempSubsystems) {
302  IndexList CurrentA(std::get<0>(x));
303  IndexList CurrentB(std::get<1>(x));
304  IndexList CurrentRecoil(std::get<2>(x));
305  IndexList CurrentParentRecoil(std::get<3>(x));
306  auto SortedEntry = sortSubsystem(x);
307  auto result = std::find_if(
308  AllSubsystems.begin(), AllSubsystems.end(),
309  [&SortedEntry, this](const IndexListTuple &element) -> bool {
310  return sortSubsystem(element) == SortedEntry;
311  });
312  if (result == AllSubsystems.end()) {
313  AllSubsystems.push_back(SortedEntry);
314  }
315 
316  // create more combinations on this level
317  for (auto const &ABnew : redistributeIndexLists(CurrentA, CurrentB)) {
318  CurrentSubsystems.push_back(std::make_tuple(
319  ABnew.first, ABnew.second, CurrentRecoil, CurrentParentRecoil));
320  }
321  // Try to go a level deeper for A
322  if (CurrentA.size() > 1) {
323  for (auto const &ABnew :
324  redistributeIndexLists(CurrentA, IndexList())) {
325  // use current B as recoil
326  // and shift recoil to parent recoil
327  CurrentSubsystems.push_back(std::make_tuple(ABnew.first, ABnew.second,
328  CurrentB, CurrentRecoil));
329  }
330  }
331  // same for B
332  if (CurrentB.size() > 1) {
333  for (auto const &ABnew :
334  redistributeIndexLists(CurrentB, IndexList())) {
335  CurrentSubsystems.push_back(std::make_tuple(ABnew.first, ABnew.second,
336  CurrentA, CurrentRecoil));
337  }
338  }
339  }
340  }
341  for (auto const &x : AllSubsystems) {
342  registerSubSystem(std::get<0>(x), std::get<1>(x), std::get<2>(x),
343  std::get<3>(x));
344  }
345 }

395 std::pair<double, double> HelicityKinematics::calculateInvMassBounds(
396  const IndexList &FinalStateIDs) const {
399  std::pair<double, double> lim(0.0, 0.0);
400  // Sum up masses of all final state particles
401  lim.first = KinematicsInfo.calculateFinalStateIDMassSum(FinalStateIDs);
402 
403  double S = KinematicsInfo.getInitialStateInvariantMassSquared();
404  double RemainderMass(0.0);
405  for (auto x : KinematicsInfo.getFinalStateMasses())
406  RemainderMass += x;
407  RemainderMass -= lim.first;
408 
409  lim.first *= lim.first;
410  // to improve precision: (M - m)^2 -> S - 2 sqrt(S) m + m^2 with S=M^2
411  lim.second =
412  S - 2 * std::sqrt(S) * RemainderMass + RemainderMass * RemainderMass;
413 
414  // extend the invariant mass interval by the numeric double precision
415  // otherwise quite a few events at the phase space boundary can be lost
416  // due to numerical imprecision of event generators
417  // (especially drastic effect, if gammas are in the final state)
418  lim.first -= std::numeric_limits<double>::epsilon() * lim.first;
419  lim.second += std::numeric_limits<double>::epsilon() * lim.second;
420  return lim;
421 }
