71 std::vector<ComPWA::Event> readData(const std::string &InputFilePath,
72  long long NumberEventsToRead) {
74  auto InputStream = openStream(InputFilePath);
75 
77  auto PIDs = extractHeader(InputStream);
78  auto NumberOfParticles = PIDs.size();
79  if (!PIDs.size())
80  throw ComPWA::BadConfig("Input data file " + InputFilePath +
81  " misses a header with PIDs");
82 
84  auto Position = InputStream.tellg();
85  bool HasWeights = false;
86  double weight, px, py, pz, e;
87  std::string line;
88  while (std::getline(InputStream, line)) {
89  if (isEmptyString(line))
90  continue;
91  std::stringstream StringStream(line);
92  if ((StringStream >> weight) && !(StringStream >> py))
93  HasWeights = true;
94  break;
95  }
96  InputStream.seekg(Position);
97 
99  bool IsOrderEnergyMomentum = false;
100  if (HasWeights)
101  InputStream >> weight;
102  InputStream >> e >> px >> py >> pz;
103  if (e * e - px * px - py * py - pz * pz > 0.)
104  IsOrderEnergyMomentum = true;
105  InputStream.seekg(Position);
106 
108  std::vector<ComPWA::Event> Events;
109  weight = 1.;
110  while (InputStream.good()) {
111  if (HasWeights)
112  InputStream >> weight;
113  std::vector<Particle> ParticleList;
114  ParticleList.reserve(NumberOfParticles);
115  for (size_t i = 0; i < NumberOfParticles; ++i) {
116  if (IsOrderEnergyMomentum)
117  InputStream >> e >> px >> py >> pz;
118  else
119  InputStream >> px >> py >> pz >> e;
120  ParticleList.push_back(ComPWA::Particle(px, py, pz, e, PIDs[i]));
121  }
122  if (!InputStream.fail())
123  Events.push_back({ParticleList, weight});
124  if (NumberEventsToRead > 0 &&
125  Events.size() == (std::size_t)NumberEventsToRead)
126  break;
127  }
128  InputStream.close();
129  return Events;
130 }
