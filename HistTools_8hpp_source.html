38  return 0;
39  }
40  TH1 *pullHist = (TH1 *)hist1->Clone(name + hist1->GetName());
41  pullHist->Reset();
42  double limitPull = 0;
43  for (int i = 1; i <= hist1->GetNbinsX(); ++i) {
44  for (int j = 1; j <= hist1->GetNbinsY(); ++j) {
45  for (int k = 1; k <= hist1->GetNbinsZ(); ++k) {
46  unsigned int bin = hist1->GetBin(i, j, k);
47  double c1 = hist1->GetBinContent(bin);
48  double c2 = hist2->GetBinContent(bin);
49  double c1Err = hist1->GetBinError(bin);
50  double c2Err = hist2->GetBinError(bin);
51  double pull;
52  if (c1 > 0 && c2 > 0) {
53  pull = (c1 - c2) / sqrt(c1Err * c1Err + c2Err * c2Err);
54  pullHist->SetBinContent(bin, pull);
55  if (std::fabs(pull) > limitPull)
56  limitPull = std::abs(pull);
57  } else
58  pullHist->SetBinContent(bin,
59  -999); // empty bins are set to error value
60  pullHist->SetBinError(bin, .0001);
61  }
62  }
63  }
64  limitPull = 4; // set fix limits
65  limitPull += 1;
66  // X
67  pullHist->GetXaxis()->SetTitleSize(.14);
68  pullHist->GetXaxis()->SetTitleOffset(.93);
69  pullHist->GetXaxis()->SetLabelSize(.12);
