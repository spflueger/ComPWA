27 std::vector<ComPWA::Event> readData(const std::string &InputFilePath,
28  const std::string &TreeName,
29  long long NumberOfEventsToRead) {
30  // Ignore custom streamer warning and error message for missing trees
31  auto temp_ErrorIgnoreLevel = gErrorIgnoreLevel;
32  gErrorIgnoreLevel = kBreak;
33 
34  // Use TChain to add files through wildcards if necessary
35  TChain TheChain(TreeName.c_str());
36  TheChain.Add(InputFilePath.c_str());
37 
38  // Test TChain quality
39  auto ListOfFiles = TheChain.GetListOfFiles();
40  if (!ListOfFiles || !ListOfFiles->GetEntries())
41  throw ComPWA::BadConfig("Root::readData() | Unable to load files: " +
42  InputFilePath);
43  if (!TheChain.GetEntries())
44  throw ComPWA::CorruptFile("Root::readData() | TTree \"" + TreeName +
45  "\" cannot be opened from file(s) " +
46  InputFilePath + "!");
47  if (NumberOfEventsToRead <= 0 || NumberOfEventsToRead > TheChain.GetEntries())
48  NumberOfEventsToRead = TheChain.GetEntries();
49  if (!TheChain.GetBranch("Particles") || !TheChain.GetBranch("weight"))
50  throw ComPWA::CorruptFile("Root::readData() | TTree does not have a "
51  "Particles and/or weight branch");
52 
53  // Set branch addresses
54  TClonesArray Particles("TParticle");
55  TClonesArray *ParticlesPointer(&Particles);
56  double Weight;
57  TheChain.GetBranch("Particles")->SetAutoDelete(false);
58  TheChain.SetBranchAddress("Particles", &ParticlesPointer);
59  TheChain.SetBranchAddress("weight", &Weight);
60 
61  std::vector<ComPWA::Event> Events;
62  Events.reserve(NumberOfEventsToRead);
63 
64  for (Long64_t i = 0; i < NumberOfEventsToRead; ++i) {
65  Particles.Clear();
66  TheChain.GetEntry(i);
67 
68  Event TheEvent;
69 
70  auto NumberOfParticles = Particles.GetEntries();
71  for (auto part = 0; part < NumberOfParticles; part++) {
72  auto TheParticle = dynamic_cast<TParticle *>(Particles.At(part));
73  if (!TheParticle)
74  continue;
75  TheEvent.ParticleList.push_back(
76  Particle(TheParticle->Px(), TheParticle->Py(), TheParticle->Pz(),
77  TheParticle->Energy(), TheParticle->GetPdgCode()));
78  } // end particle loop
79  TheEvent.Weight = Weight;
80 
81  Events.push_back(TheEvent);
82  } // end event loop
83 
84  gErrorIgnoreLevel = temp_ErrorIgnoreLevel;
85  return Events;
86 }

88 void writeData(const std::vector<ComPWA::Event> &Events,
89  const std::string &OutputFileName, const std::string &TreeName,
90  bool OverwriteFile) {
91  // Ignore custom streamer warning
92  auto temp_ErrorIgnoreLevel = gErrorIgnoreLevel;
93  gErrorIgnoreLevel = kBreak;
94 
95  if (0 == Events.size()) {
96  LOG(ERROR) << "Root::writeData(): no events given!";
97  return;
98  }
99 
100  LOG(INFO) << "Root::writeData(): writing vector of " << Events.size()
101  << " events to file " << OutputFileName;
102 
103  std::string WriteFlag{"UPDATE"};
104  if (OverwriteFile)
105  WriteFlag = "RECREATE";
106 
107  TFile TheFile(OutputFileName.c_str(), "UPDATE");
108  if (TheFile.IsZombie())
109  throw std::runtime_error("Root::writeData(): can't open data file: " +
110  OutputFileName);
111 
112  // TTree branch variables
113  TTree TheTree(TreeName.c_str(), TreeName.c_str());
114  auto ParticlesPointer =
115  new TClonesArray("TParticle", Events[0].ParticleList.size());
116  double Weight;
117  TheTree.Branch("Particles", &ParticlesPointer);
118  TheTree.Branch("weight", &Weight, "weight/D");
119  auto &Particles = *ParticlesPointer;
120 
121  for (auto const &Event : Events) {
122  Particles.Clear();
123  Weight = Event.Weight;
124  double Mass = ComPWA::calculateInvariantMass(Event);
125  TLorentzVector motherMomentum(0, 0, 0, Mass);
126  for (unsigned int i = 0; i < Event.ParticleList.size(); ++i) {
127  auto &Particle = Event.ParticleList[i];
128  auto FourMom(Particle.fourMomentum());
129  TLorentzVector oldMomentum(FourMom.px(), FourMom.py(), FourMom.pz(),
130  FourMom.e());
131  Particles[i] = new TParticle(Particle.pid(), 1, 0, 0, 0, 0, oldMomentum,
132  motherMomentum);
133  // TClonesArray owns its objects
134  }
135  TheTree.Fill();
136  }
137  TheTree.Write("", TObject::kOverwrite, 0);
138  TheFile.Close();
139  gErrorIgnoreLevel = temp_ErrorIgnoreLevel;
140 }
