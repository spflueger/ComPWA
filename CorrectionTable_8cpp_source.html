102 double CorrectionTable::GetValue(int charge, double momentum) const {
103  int binNumber = findBin(momentum);
104  if (binNumber < 0)
105  throw std::runtime_error(
106  "momentumSys::value() no bin found for momentum value " +
107  std::to_string((long double)momentum) + "!");
108 
109  double val = 0.0;
110  if (charge > 0)
111  val = sys.at(binNumber);
112  else if (charge < 0)
113  val = antiSys.at(binNumber);
114  else if (charge == 0) { // charge unknown -> average value
115  val = (antiSys.at(binNumber) + sys.at(binNumber)) / 2;
116  }
117  return val;
118 }
119 double CorrectionTable::GetError(int charge, double momentum) const {
120  int binNumber = findBin(momentum);
121  if (binNumber < 0)
122  throw std::runtime_error(
123  "momentumSys::GetError() no bin found for momentum value " +
124  std::to_string((long double)momentum) + "!");
125 
126  double err = 0.0;
127  if (charge > 0) // D0->K0bar K+K-
128  err = sysError.at(binNumber);
129  else if (charge < 0) // D0->K0bar K+K-
130  err = antiSysError.at(binNumber);
131  else if (charge == 0) { // charge unknown -> average value
132  err = (0.5 *
133  std::sqrt(antiSysError.at(binNumber) * antiSysError.at(binNumber) +
134  sysError.at(binNumber) * sysError.at(binNumber)));
135  }
136  return err;
137 }

150 double CorrectionTable::GetTotalSystematics(bool useArithmeticMean) {
151  double mean = 0;
152  double sumSigma = 0;
153  // use arithmetic mean
154  if (useArithmeticMean) {
155  for (unsigned int i = 0; i < totalSys.size(); i++)
156  mean += totalSys.at(i);
157  return mean / totalSys.size();
158  } else {
159  // use weighted mean
160  for (unsigned int i = 0; i < totalSys.size(); i++) {
161  double tmp = 1 / (totalSysError.at(i) * totalSysError.at(i));
162  mean += tmp * totalSys.at(i);
163  sumSigma += tmp;
164  }
165  mean /= sumSigma;
166  return mean;
167  }
168 }

170 double CorrectionTable::GetTotalSystematicsError(bool useArithmeticMean) {
171  double sumSigma = 0; // inverse sum over uncertainties
172  double mean = GetTotalSystematics(useArithmeticMean);
173  if (useArithmeticMean) {
174  for (unsigned int i = 0; i < totalSys.size(); i++)
175  sumSigma += (totalSys.at(i) - mean) * (totalSys.at(i) - mean);
176  return sumSigma = sqrt(sumSigma / totalSys.size());
177  } else {
178  for (unsigned int i = 0; i < totalSysError.size(); i++)
179  sumSigma += 1 / (totalSysError.at(i) * totalSysError.at(i));
180  return sqrt(1 / sumSigma);
181  }
182 }
