54 std::vector<ComPWA::Event> DalitzKinematics::reduceToPhaseSpace(
55  const std::vector<ComPWA::Event> &Events) const {
56  std::vector<Event> Evts;
57 
58  auto Dataset = convert(Events);
59  auto mA = Dataset.Data["mA"];
60  auto mB = Dataset.Data["mB"];
61  auto mC = Dataset.Data["mC"];
62  auto qAB = Dataset.Data["qAB"];
63  auto qBC = Dataset.Data["qBC"];
64  auto qCA = Dataset.Data["qCA"];
65 
66  for (size_t i = 0; i < Events.size(); ++i) {
67  double s2 = (qAB[i] + qBC[i] + qCA[i] - mA[i] * mA[i] - mB[i] * mB[i] -
68  mC[i] * mC[i]);
69 
70  if (s2 < M2)
71  Evts.push_back(Events[i]);
72  }
73  return Evts;
74 }

76 ComPWA::Data::DataSet
77 DalitzKinematics::convert(const std::vector<Event> &Events) const {
78 
79  ComPWA::Data::DataSet Dataset;
80 
81  std::vector<double> mA, mB, mC, qAB, qBC, qCA, weights;
82  for (auto const &event : Events) {
83  mA.push_back(event.ParticleList[0].mass());
84  mB.push_back(event.ParticleList[1].mass());
85  mC.push_back(event.ParticleList[2].mass());
86  qAB.push_back((event.ParticleList[0].fourMomentum() +
87  event.ParticleList[1].fourMomentum())
88  .invariantMass());
89  qBC.push_back((event.ParticleList[1].fourMomentum() +
90  event.ParticleList[2].fourMomentum())
91  .invariantMass());
92  qCA.push_back((event.ParticleList[2].fourMomentum() +
93  event.ParticleList[0].fourMomentum())
94  .invariantMass());
95  weights.push_back(event.Weight);
96  }
97 
98  Dataset.Data.insert(std::make_pair("mA", mA));
99  Dataset.Data.insert(std::make_pair("mB", mB));
100  Dataset.Data.insert(std::make_pair("mC", mC));
101  Dataset.Data.insert(std::make_pair("qAB", qAB));
102  Dataset.Data.insert(std::make_pair("qBC", qBC));
103  Dataset.Data.insert(std::make_pair("qCA", qCA));
104  Dataset.Data.insert(std::make_pair("weights", weights));
105 
106  return Dataset;
107 }
