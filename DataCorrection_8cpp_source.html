17 double MomentumCorrection::correction(Event &ev) const {
18  double w = 1;
19  for (unsigned int i = 0; i < ev.ParticleList.size(); ++i) {
20  Particle p = ev.ParticleList[i];
21  int charge =
22  findParticle(PartList, p.pid()).getQuantumNumber<int>("charge");
23  double mom = p.threeMomentum();
24  double corr;
25  try {
26  corr = Corrections.at(i).GetValue(charge, mom) + 1;
27  } catch (std::exception &ex) {
28  throw std::runtime_error(
29  "MomentumCorrection::getCorrection() | Number of histograms is "
30  "expected to be number of final state particles!");
31  }
32  w *= corr;
33  }
34  return w;
35 }
