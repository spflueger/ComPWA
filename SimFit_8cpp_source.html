197 energyPar createEnergyPar(size_t NEvents, double SqrtS, int seed,
198  ComPWA::ParticleList partL,
199  std::vector<ComPWA::pid> initialState,
200  std::vector<ComPWA::pid> finalState,
201  const boost::property_tree::ptree &ModelPT) {
202  auto kin = ComPWA::Physics::HelicityFormalism::HelicityKinematics(
203  partL, initialState, finalState,
204  ComPWA::FourMomentum(0.0, 0.0, 0.0, SqrtS));
205 
206  ComPWA::Data::Root::RootGenerator gen(
207  kin.getParticleStateTransitionKinematicsInfo());
208 
209  ComPWA::Data::Root::RootUniformRealGenerator RandomGenerator(seed);
210 
211  auto mcSample = ComPWA::Data::generatePhsp(100000, gen, RandomGenerator);
212 
213  // new builder because we need to use a different phsp sample for
214  // normalization
215  auto Builder = ComPWA::Physics::IntensityBuilderXML(
216  partL, kin, ModelPT.get_child("Intensity"), mcSample);
217  // Construct intensity class from model string
218  auto amp = Builder.createIntensity();
219 
220  auto data =
221  ComPWA::Data::generate(NEvents, kin, RandomGenerator, amp, mcSample);
222  auto dataSet = kin.convert(data);
223  return energyPar{NEvents, std::move(kin), std::move(amp),
224  data, mcSample, dataSet};
225 }
