233 int main(int argc, char **argv) {
234  // initialize logging
235  Logging log("debug", "DalitzFit-log.txt");
236 
237  // List with all particle information needed
238  std::stringstream ParticlesStream(myParticles);
239  ParticleList partL = readParticles(ParticlesStream);
240 
241  //---------------------------------------------------
242  // 1) Create Kinematics object
243  //---------------------------------------------------
244  std::vector<pid> initialState = {443};
245  std::vector<pid> finalState = {22, 111, 111};
246  HelicityKinematics kin(partL, initialState, finalState);
247 
248  //---------------------------------------------------
249  // 2) Generate a large phase space sample
250  //---------------------------------------------------
251  ComPWA::Data::Root::RootGenerator gen(
252  kin.getParticleStateTransitionKinematicsInfo());
253 
254  ComPWA::Data::Root::RootUniformRealGenerator RandomGenerator(173);
255 
256  auto phspSample(ComPWA::Data::generatePhsp(100000, gen, RandomGenerator));
257 
258  //---------------------------------------------------
259  // 3) Create intensity from pre-defined model
260  //---------------------------------------------------
261  // Read in model property_tree
262  std::stringstream modelStream;
263  modelStream << amplitudeModel;
264  boost::property_tree::ptree modelTree;
265  boost::property_tree::xml_parser::read_xml(modelStream, modelTree);
266 
267  // Construct intensity class from model string
268  ComPWA::Physics::IntensityBuilderXML Builder(
269  partL, kin, modelTree.get_child("Intensity"), phspSample);
270  auto intens = Builder.createIntensity();
271 
272  //---------------------------------------------------
273  // 4) Generate a data sample given intensity and kinematics
274  //---------------------------------------------------
275  auto sample =
276  ComPWA::Data::generate(1000, kin, RandomGenerator, intens, phspSample);
277 
278  auto SampleDataSet = kin.convert(sample);
279  //---------------------------------------------------
280  // 5) Fit the model to the data and print the result
281  //---------------------------------------------------
282  auto esti = ComPWA::Estimator::createMinLogLHFunctionTreeEstimator(
283  intens, SampleDataSet);
284 
285  LOG(DEBUG) << esti.first.print(25);
286 
287  auto minuitif = Optimizer::Minuit2::MinuitIF();
288 
289  // STARTING MINIMIZATION
290  auto result = minuitif.optimize(std::get<0>(esti), std::get<1>(esti));
291 
292  LOG(INFO) << result;
293 
294  // calculate fit fractions and errors
295  auto components = Builder.createIntensityComponents(
296  {{"f2(1270)"}, {"myAmp"}, {"jpsiGammaPiPi"}});
297 
298  auto MyFractions = {std::make_pair(components[0], components[2]),
299  std::make_pair(components[1], components[2])};
300 
301  ComPWA::Tools::FitFractions FF;
302  auto FitFractionList = FF.calculateFitFractionsWithCovarianceErrorPropagation(
303  MyFractions, kin.convert(phspSample), result);
304 
305  LOG(INFO) << FitFractionList;
306 
307  //---------------------------------------------------
308  // 5.1) Save the fit result
309  //---------------------------------------------------
310  std::ofstream ofs("DalitzFit-fitResult.xml");
311  boost::archive::xml_oarchive oa(ofs);
312  oa << BOOST_SERIALIZATION_NVP(result);
313 
314  //---------------------------------------------------
315  // 6) Plot data sample and intensity
316  //---------------------------------------------------
317  ComPWA::Tools::Plotting::DalitzPlot pl(kin, "DalitzFit", 100);
318  pl.fill(sample, true, "data", "Data sample", kBlack);
319  pl.fill(phspSample, false, "phsp", "Phsp sample", kGreen);
320  pl.fill(phspSample, intens, false, "fit", "jpsiGammaPiPi model", kBlue);
321  pl.plot();
322  LOG(INFO) << "Done";
323 
324  return 0;
325 }
