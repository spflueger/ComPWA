112 ComPWA::Event RootGenerator::generate(UniformRealNumberGenerator &gen) const {
113  ComPWA::Event evt;
114 
115  size_t NumberOfFinalStateParticles(FinalStateMasses.size());
116  std::vector<double> OrderedRandomNumbers;
117  OrderedRandomNumbers.reserve(NumberOfFinalStateParticles);
118  OrderedRandomNumbers.push_back(0.0);
119 
120  if (NumberOfFinalStateParticles > 2) {
121  for (unsigned int n = 1; n < NumberOfFinalStateParticles - 1; ++n)
122  OrderedRandomNumbers.push_back(gen()); // N-2 random numbers
123  std::sort(OrderedRandomNumbers.begin(), OrderedRandomNumbers.end());
124  }
125  OrderedRandomNumbers.push_back(1.0);
126 
127  double invMas[NumberOfFinalStateParticles], sum = 0.0;
128  for (unsigned int n = 0; n < NumberOfFinalStateParticles; ++n) {
129  sum += FinalStateMasses[n];
130  invMas[n] = OrderedRandomNumbers[n] * CMSEnergyMinusMasses + sum;
131  }
132 
133  // compute the weight of the current event
134  double weight = MaximumWeight;
135  double pd[NumberOfFinalStateParticles];
136  for (unsigned int n = 0; n < NumberOfFinalStateParticles - 1; ++n) {
137  pd[n] = PDK(invMas[n + 1], invMas[n], FinalStateMasses[n + 1]);
138  weight *= pd[n];
139  }
140 
141  std::vector<TLorentzVector> FinalStateLorentzVectors(FinalStateMasses.size());
142 
143  // complete specification of event (Raubold-Lynch method)
144  FinalStateLorentzVectors[0].SetPxPyPzE(
145  0.0, pd[0], 0.0,
146  std::sqrt(std::pow(pd[0], 2) + std::pow(FinalStateMasses[0], 2)));
147 
148  unsigned int i(1);
149  while (true) {
150  FinalStateLorentzVectors[i].SetPxPyPzE(
151  0.0, -pd[i - 1], 0.0,
152  std::sqrt(std::pow(pd[i - 1], 2) + std::pow(FinalStateMasses[i], 2)));
153 
154  double cZ = 2.0 * gen() - 1.0;
155  double sZ = std::sqrt(1.0 - std::pow(cZ, 2));
156  double angY = 2.0 * TMath::Pi() * gen();
157  double cY = std::cos(angY);
158  double sY = std::sin(angY);
159  for (unsigned int j = 0; j <= i; ++j) {
160  TLorentzVector &v(FinalStateLorentzVectors[j]);
161  double x = v.Px();
162  double y = v.Py();
163  double z = v.Pz();
164  // rotation around Z and Y
165  v.SetPx(cY * (cZ * x - sZ * y) - sY * z);
166  v.SetPy(sZ * x + cZ * y);
167  v.SetPz(sY * (cZ * x - sZ * y) + cY * z);
168  }
169  if (i == NumberOfFinalStateParticles - 1)
170  break;
171  // double beta = 1.0 / std::sqrt(1.0 + std::pow(invMas[i] / pd[i], 2));
172  double beta2 = 1.0 / (1.0 + std::pow(invMas[i] / pd[i], 2));
173  for (unsigned int j = 0; j <= i; ++j) {
174  // FinalStateLorentzVectors[j].Boost(0.0, beta, 0.0);
175  BoostAlongY(FinalStateLorentzVectors[j], beta2);
176  }
177  ++i;
178  }
179 
180  // final boost of all particles
181  for (unsigned int n = 0; n < NumberOfFinalStateParticles; ++n) {
182  FinalStateLorentzVectors[n].Boost(CMSBoostVector);
183  evt.ParticleList.push_back(Particle(
184  FinalStateLorentzVectors[n].X(), FinalStateLorentzVectors[n].Y(),
185  FinalStateLorentzVectors[n].Z(), FinalStateLorentzVectors[n].E(), 0));
186  }
187  evt.Weight = weight;
188  return evt;
189 }

191 void RootGenerator::BoostAlongY(TLorentzVector &vec,
192  double beta_squared) const {
193  // Boost this Lorentz vector
194  double y(vec.Y());
195  double t(vec.T());
196 
197  double gamma = 1.0 / std::sqrt(1.0 - beta_squared);
198  double betagamma = 1.0 / std::sqrt((1.0 - beta_squared) / beta_squared);
199 
200  vec.SetY(gamma * y + betagamma * t);
201  vec.SetT(gamma * t + betagamma * y);
202 }

204 double RootGenerator::PDK(double a, double b, double c) const {
205  double x = (a - b - c) * (a + b + c) * (a - b + c) * (a + b - c);
206  return std::sqrt(x) / (2.0 * a);
207 }
