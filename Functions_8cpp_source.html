95 KahanSummation KahanSum(KahanSummation accumulation, double value) {
96  KahanSummation result;
97  double y = value - accumulation.correction;
98  double t = accumulation.sum + y;
99  result.correction = (t - accumulation.sum) - y;
100  result.sum = t;
101  return result;
