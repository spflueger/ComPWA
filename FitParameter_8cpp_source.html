64 void FitParameter::updateParameter(std::shared_ptr<FitParameter> newPar) {
65 
66  // Copy bounds
67  if (newPar->hasBounds()) {
68  HasBounds = 1;
69  setBounds(newPar->bounds());
70  } else
71  HasBounds = 0;
72 
73  bool isFix = newPar->isFixed();
74  fixParameter(0); // we ignore here if parameter is fixed
75 
76  // Copy value
77  setValue(newPar->value());
78 
79  // Copy uncertainty
80  if (newPar->errorType() == ErrorType::SYM)
81  setError(newPar->error().first);
82  else if (newPar->errorType() == ErrorType::ASYM)
83  setError(newPar->error());
84  else
85  ErrType = ErrorType::NOTDEF;
86 
87  // Copy fix parameter
88  fixParameter(isFix);
89  return;
90 }

92 void FitParameter::setValue(const double inVal) {
93  if (IsFixed)
94  throw ParameterFixed("FunctionTree::FitParameter::setValue() | Parameter " +
95  name() + " is fixed!");
96  // Call notify only if value has changed! Otherwise tree is
97  // recalculated also in case where current parameter is not changed
98  if (Value == inVal)
99  return;
100 
101  if (HasBounds && (inVal < bounds().first || inVal > bounds().second))
102  throw ParameterOutOfBound(
103  "FitParameter::setValue() | Parameter " + name() +
104  " not within bounds: val=" + std::to_string(inVal) + " [" +
105  std::to_string(bounds().first) + ";" + std::to_string(bounds().second) +
106  "]");
107 
108  Value = inVal;
109  notify();
110 }

172 bool FitParameter::operator==(const FitParameter otherPar) const {
173  if (this->type() != otherPar.type())
174  return false;
175  if (this->name() != otherPar.name())
176  return false;
177  if (this->value() != otherPar.value())
178  return false;
179 
180  // We assume that if name and value are the same both parameters match. In
181  // case that other properties of the parameterts differ we throw an
182  // exception since we assume that this is a user mistake.
183  if (HasBounds != otherPar.HasBounds || this->bounds() != otherPar.bounds())
184  throw std::runtime_error("FitParameter::operator==() | Parameters "
185  "match by name (" +
186  name() + ") and value (" +
187  std::to_string(value()) +
188  ") but differs in "
189  "parameter bounds. We assume that there is a "
190  "mistake. Check your input files!");
191 
192  if (IsFixed != otherPar.IsFixed)
193  throw std::runtime_error(
194  "FitParameter::operator==() | Parameters "
195  "match by name (" +
196  name() + ") and value (" + std::to_string(value()) +
197  ") but one is fixed the other not. "
198  "We assume that there is a mistake. Check your input files!");
199 
200  if (ErrType != otherPar.ErrType || Error != otherPar.Error)
201  throw std::runtime_error("FitParameter::operator==() | Parameters "
202  "match by name (" +
203  name() + ") and value (" +
204  std::to_string(value()) +
205  ") but differs in "
206  "parameter error. We assume that there is a "
207  "mistake. Check your input files!");
208 
209  return true;
210 }

242 void FitParameter::load(const boost::property_tree::ptree &pt) {
243 
244  // Class attribute is not required. But if it is specified we expect 'Double'
245  // here.
246  auto c = pt.get_optional<std::string>("<xmlattr>.Class");
247  if (c && c.get() != "Double")
248  throw BadConfig("CoherentIntensity::Factory() | Property tree seems to "
249  "not containt a configuration for an CoherentIntensity!");
250 
251  // Require that name and value are provided
252  Name = pt.get<std::string>("<xmlattr>.Name");
253  Value = pt.get<double>("Value");
254 
255  // Optional settings
256  if (pt.get_optional<double>("Error")) {
257  this->setError(pt.get<double>("Error"));
258  }
259  if (pt.get_optional<double>("ErrorLow")) {
260  if (pt.get_optional<double>("ErrorHigh"))
261  this->setError(std::pair<double, double>(pt.get<double>("ErrorLow"),
262  pt.get<double>("ErrorHigh")));
263  else
264  throw std::runtime_error("FitParameterFactory() | Parameter asymmetic "
265  "error not properly set!");
266  } else if (pt.get_optional<double>("ErrorHigh")) {
267  throw std::runtime_error("FitParameterFactory() | Parameter asymmetic "
268  "error not properly set!");
269  } else { // Do not set a asymmetric errors
270  }
271 
272  auto fix = pt.get_optional<bool>("Fix");
273  if (fix)
274  this->fixParameter(fix.get());
275  else
276  this->fixParameter(true);
277 
278  auto min = pt.get_optional<double>("Min");
279  auto max = pt.get_optional<double>("Max");
280  if (min && max) {
281  this->setBounds(min.get(), max.get());
282  } else if (min || max) { // Bounds not completely specified
283  throw std::runtime_error(
284  "FitParameterFactory() | Parameter bounds not properly set!");
285  } else { // No bounds specified
286  }
287 }
