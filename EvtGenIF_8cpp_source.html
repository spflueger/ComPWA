195 std::vector<double> EvtGenIF::evaluate(const ComPWA::DataMap &data) noexcept {
196  std::vector<double> Results;
197  for (size_t EventIndex = 0; EventIndex < data.at("mA").size(); ++EventIndex) {
198  EvtDalitzPoint pnt(data.at("mA")[EventIndex], data.at("mB")[EventIndex],
199  data.at("mC")[EventIndex], data.at("qAB")[EventIndex],
200  data.at("qBC")[EventIndex], data.at("qCA")[EventIndex]);
201  double result = 0;
202 
203  for (unsigned int i = 0; i < Resos.size(); ++i) {
204  EvtDalitzReso tmp = Resos.at(i);
205  std::string name = tmp.get_Name();
206 
207  tmp.set_Mass(evtPars.at(std::string(name + "_mass"))->value());
208  tmp.set_Gamma(evtPars.at(std::string(name + "_width"))->value());
209 
210  result += abs2(tmp.evaluate(pnt)); // * normValues.at(i);
211  }
212 
213  assert(!std::isnan(result) &&
214  "IncoherentIntensity::Intensity() | Result is NaN!");
215  assert(!std::isinf(result) &&
216  "IncoherentIntensity::Intensity() | Result is inf!");
217 
218  Results.push_back(result);
219  }
220  return Results;
