261 void DalitzHisto::fill(const Data::DataSet &sample,
262  std::vector<double> Intensities) {
263  if (Intensities.size() == 0) {
264  Intensities = std::vector<double>(sample.Weights.size(), 1.0);
265  }
266  if (!sample.Data.size())
267  throw std::runtime_error("DalitzHist::fill() | Empty data sample.");
268  if (Intensities.size() != sample.Weights.size())
269  throw std::runtime_error("DalitzHist::fill() | Vector of weights and "
270  "sample do not have the same length");
271 
272  double weight =
273  std::accumulate(sample.Weights.begin(), sample.Weights.end(), 0.0);
274 
275  Integral += weight;
276 
277  std::vector<double> weights;
278  for (size_t i = 0; i < Intensities.size(); ++i) {
279  weights.push_back(sample.Weights[i] * Intensities[i]);
280  }
281 
282  for (auto &h1 : Hists1D) {
283  auto const &data = sample.Data.at(h1.first);
284  for (size_t i = 0; i < data.size(); ++i) {
285  h1.second.Fill(data[i], weights[i]);
286  }
287  }
288 
289  for (auto &h2 : Hists2D) {
290  auto const &data1 = sample.Data.at(h2.first.first);
291  auto const &data2 = sample.Data.at(h2.first.second);
292  for (size_t i = 0; i < data1.size(); ++i) {
293  h2.second.Fill(data1[i], data2[i], weights[i]);
294  }
295  }
296 
297  auto const &m23sq = sample.Data.at("mSq_(1,2)");
298  auto const &m13sq = sample.Data.at("mSq_(0,2)");
299  auto const &m12sq = sample.Data.at("mSq_(0,1)");
300  for (size_t i = 0; i < weights.size(); ++i) {
301  BranchPoint = {m23sq[i], m13sq[i], m12sq[i]};
302  BranchWeight = weights[i];
303  BranchEff = 1.0;
304  Tree->Fill();
305  }
306 }
